#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

int main() {
    const int m = 3;
    const int n = 4;
    int mas[m][n];

    srand(time(0));

    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            mas[i][j] = rand() % 100;
            cout << mas[i][j] << " ";
        }
        cout << endl;
    }

    int maxVal = mas[0][0];
    int maxRow = 0;
    int maxCol = 0;

    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            if (mas[i][j] > maxVal) {
                maxVal = mas[i][j];
                maxRow = i;
                maxCol = j;
            }
        }
    }

    cout << "max value = " << maxVal << endl;
    cout << "position: row = " << maxRow << ", column = " << maxCol << endl;

    return 0;
}
