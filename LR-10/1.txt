#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

int main() {
    const int m = 4;
    const int n = 3;
    int mas[m][n];

    srand(time(0));

    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            mas[i][j] = rand() % 100;
            cout << mas[i][j] << " ";
        }
        cout << endl;
    }

    for (int i = 0; i < m; i++) {
        int k = 0;
        for (int j = 0; j < n; j++) {
            if (mas[i][j] % 2 == 0) {
                k++;
            }
        }
        cout << "row " << i << " even count = " << k << endl;
    }

    return 0;
}
