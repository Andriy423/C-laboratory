#include <iostream>
using namespace std;

int main()
{
    int N, count = 0, digit;
    int temp;
    
    cout << "Enter natural number N: ";
    cin >> N;
    
    if (N <= 0) {
        cout << "Error: N must be a natural number (N > 0)" << endl;
        return 1;
    }
    
    temp = N;
    
    do {
        digit = temp % 10;
        
        if (digit != 1 && digit != 9) {
            count++;
        }
        
        temp = temp / 10;
        
    } while (temp > 0);
    
    cout << endl;
    cout << "Count of digits not equal to 1 and 9: " << count << endl;
}
