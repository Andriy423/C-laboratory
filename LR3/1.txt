#include <iostream>
#include <cmath>
using namespace std;


// Завдання 1
int main()
{
    int number, sum, digit_one, digit_two, digit_three;
    sum = 1;
    
    cout << "Enter a three-digit number: ";
    cin >> number;
    
    digit_one = number / 100;
    digit_two = number / 10 % 10;
    digit_three = number % 10;
    
    if (digit_one % 2 == 1){
        sum *= digit_one;
    } if(digit_two % 2 == 1) {
        sum *= digit_two;
    } if(digit_three % 2 == 1) {
        sum *= digit_three;
    }
    
    cout << "SUM = "<< sum;
    
}
