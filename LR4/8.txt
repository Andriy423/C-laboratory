#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    double first_number, second_number;
    const double EPSILON = 0.000001;
    
    cout << "Enter a number 1: ";
    cin >> first_number;
    
    cout << "Enter a number 2: ";
    cin >> second_number;
    
    double result1 = 1.0 / first_number;
    double result2 = 1.0 / second_number;
    
    if (abs(result1 - result2) < EPSILON) {
        cout << "Results are equal (by 0.000001 epsilon)";
    } else {
        cout << "Results are not equal (by 0.000001 epsilon)";
    }
    
    
    return 0;
}
