#include <iostream>
using namespace std;

int main()
{
    int num1, num2, num3, num4;
    bool valid = true;
    
    cout << "Enter first number (1-255): ";
    cin >> num1;
    
    cout << "Enter second number (1-255): ";
    cin >> num2;
    
    cout << "Enter third number (1-255): ";
    cin >> num3;
    
    cout << "Enter fourth number (1-255): ";
    cin >> num4;
    
    if (num1 < 1 || num1 > 255) {
        cout << "Error: First number " << num1 << " is out of range (1-255)" << endl;
        valid = false;
    }
    
    if (num2 < 1 || num2 > 255) {
        cout << "Error: Second number " << num2 << " is out of range (1-255)" << endl;
        valid = false;
    }
    
    if (num3 < 1 || num3 > 255) {
        cout << "Error: Third number " << num3 << " is out of range (1-255)" << endl;
        valid = false;
    }
    
    if (num4 < 1 || num4 > 255) {
        cout << "Error: Fourth number " << num4 << " is out of range (1-255)" << endl;
        valid = false;
    }
    
    if (valid) {
        cout << num1 << "." << num2 << "." << num3 << "." << num4 << endl;
    } else {
        cout << "Invalid IP address due to out of range values." << endl;
    }
    
    return 0;
}
