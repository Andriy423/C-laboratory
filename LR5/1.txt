#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

int main()
{
    double x, y;
    
    for (int i = 1; i <= 5; i++) {
        cout << "Enter x" << i << ": ";
        cin >> x;
        
        y = sin(5 * x) + 1.5 * x - 5;
        
        cout << fixed << "y = " << setprecision(6) << y << endl;
        cout << endl;
    }
}
