#include <iostream>
using namespace std;

struct Date {
    int hours = 0;
    int minutes = 0;
};

bool isValidTime(int h, int m) {
    return (h >= 0 && h < 24) && (m >= 0 && m < 60);
}

Date addMinutes(Date t, int addMin) {
    t.minutes += addMin;
    t.hours += t.minutes / 60;
    t.minutes %= 60;
    t.hours %= 24;
    return t;
}

int main() {
    Date time;
    int additionalMinutes;

    cin >> time.hours;
    cin >> time.minutes;

    if (!isValidTime(time.hours, time.minutes)) {
        cout << "Invalid time!" << endl;
        return 1;
    }

    cin >> additionalMinutes;

    Date newTime = addMinutes(time, additionalMinutes);

    cout << newTime.hours << ":";
    if (newTime.minutes < 10) cout << "0";
    cout << newTime.minutes << endl;
}
